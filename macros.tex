\newcommand{\twochoices}[2]{
    \left \{%
    \begin{array}{lcc}
        \displaystyle #1 \\
        \vspace{-10pt}   \\
        \displaystyle #2
    \end{array}
    \right.
}

\newcommand{\twovec}[2]{
    \left(%
    \begin{array}{c}
        #1 \\ #2
    \end{array}
    \right)
}

\newcommand{\twomatrix}[4]{
    \left(%
    \begin{array}{cc}
        #1 & #2 \\
        #3 & #4
    \end{array}
    \right)
}
